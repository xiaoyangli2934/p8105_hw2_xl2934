## Read and clean precipitation data from 2017 and 2018

``` r
prec_2018 = 
  read_excel(
    path = "./data/HealthyHarborWaterWheelTotals2018-7-28.xlsx", 
    sheet = "2018 Precipitation",
    range = "A2:B14") %>% 
  select(-Month) %>% 
  mutate(
    year = "2018", 
    month = month.name)

prec_2017 = 
  read_excel(
    path = "./data/HealthyHarborWaterWheelTotals2018-7-28.xlsx", 
    sheet = "2017 Precipitation",
    range = "A2:B14") %>% 
  select(-Month) %>% 
  mutate(
    year = "2017",
    month = month.name)

prec = bind_rows(prec_2017, prec_2018) %>% 
  janitor::clean_names() %>% 
  drop_na(total)
```
